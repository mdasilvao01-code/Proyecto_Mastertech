import requests
import json
import os
from datetime import datetime

# Configuración del servidor
SERVER_URL = "http://localhost:5000"  # Cambiar por tu dominio: http://tu-dominio.com

class ClienteMastertech:
    def __init__(self):
        self.usuario = None
        self.session = requests.Session()
    
    def limpiar_pantalla(self):
        os.system('clear' if os.name != 'nt' else 'cls')
    
    def mostrar_encabezado(self, titulo):
        self.limpiar_pantalla()
        print("=" * 60)
        print(f"🔧 MASTERTECH - {titulo}")
        print("=" * 60)
        if self.usuario:
            print(f"👤 Usuario: {self.usuario['nombre']} ({self.usuario['email']})")
            print("=" * 60)
        print()
    
    def login(self):
        while True:
            self.mostrar_encabezado("Inicio de Sesión - CLIENTE")
            
            print("Ingrese sus credenciales:")
            email = input("📧 Email: ").strip()
            password = input("🔒 Contraseña: ").strip()
            
            try:
                response = self.session.post(f"{SERVER_URL}/api/login", 
                    json={"email": email, "password": password})
                
                data = response.json()
                
                if data.get('success') and data['usuario']['rol'] == 'cliente':
                    self.usuario = data['usuario']
                    print(f"\n✅ Bienvenido {self.usuario['nombre']}!")
                    input("\nPresiona Enter para continuar...")
                    return True
                else:
                    print("\n❌ Error: Credenciales incorrectas o no es un cliente")
                    input("\nPresiona Enter para intentar de nuevo...")
            
            except requests.exceptions.ConnectionError:
                print(f"\n❌ Error: No se puede conectar al servidor en {SERVER_URL}")
                print("Verifica que el servidor esté ejecutándose.")
                input("\nPresiona Enter para intentar de nuevo...")
            except Exception as e:
                print(f"\n❌ Error: {e}")
                input("\nPresiona Enter para intentar de nuevo...")
    
    def menu_principal(self):
        while True:
            self.mostrar_encabezado("Menú Principal")
            
            print("1. 📝 Crear Nueva Incidencia")
            print("2. 📋 Ver Mis Incidencias")
            print("3. 🔍 Ver Detalle de Incidencia")
            print("4. 💬 Añadir Comentario a Incidencia")
            print("5. 🚪 Salir")
            print()
            
            opcion = input("Selecciona una opción: ").strip()
            
            if opcion == '1':
                self.crear_incidencia()
            elif opcion == '2':
                self.ver_mis_incidencias()
            elif opcion == '3':
                self.ver_detalle_incidencia()
            elif opcion == '4':
                self.añadir_comentario()
            elif opcion == '5':
                print("\n👋 ¡Hasta pronto!")
                break
            else:
                print("\n❌ Opción no válida")
                input("\nPresiona Enter para continuar...")
    
    def crear_incidencia(self):
        self.mostrar_encabezado("Crear Nueva Incidencia")
        
        print("Complete los siguientes datos:\n")
        
        titulo = input("📌 Título de la incidencia: ").strip()
        if not titulo:
            print("\n❌ El título no puede estar vacío")
            input("\nPresiona Enter para continuar...")
            return
        
        descripcion = input("📄 Descripción detallada: ").strip()
        if not descripcion:
            print("\n❌ La descripción no puede estar vacía")
            input("\nPresiona Enter para continuar...")
            return
        
        print("\n📁 Categorías disponibles:")
        categorias = ["Hardware", "Software", "Red", "Seguridad", 
                     "Base de Datos", "Correo Electrónico", "Otros"]
        for i, cat in enumerate(categorias, 1):
            print(f"  {i}. {cat}")
        
        cat_num = input("\nSelecciona categoría (1-7): ").strip()
        if not cat_num.isdigit() or int(cat_num) < 1 or int(cat_num) > 7:
            print("\n❌ Categoría no válida")
            input("\nPresiona Enter para continuar...")
            return
        categoria = categorias[int(cat_num) - 1]
        
        print("\n⚡ Prioridades:")
        prioridades = ["Baja", "Media", "Alta", "Crítica"]
        for i, pri in enumerate(prioridades, 1):
            print(f"  {i}. {pri}")
        
        pri_num = input("\nSelecciona prioridad (1-4): ").strip()
        if not pri_num.isdigit() or int(pri_num) < 1 or int(pri_num) > 4:
            print("\n❌ Prioridad no válida")
            input("\nPresiona Enter para continuar...")
            return
        prioridad = prioridades[int(pri_num) - 1]
        
        # Crear incidencia
        datos = {
            "titulo": titulo,
            "descripcion": descripcion,
            "categoria": categoria,
            "prioridad": prioridad,
            "cliente_id": self.usuario['id'],
            "cliente_nombre": self.usuario['nombre']
        }
        
        try:
            response = self.session.post(f"{SERVER_URL}/api/incidencias", json=datos)
            
            if response.status_code == 201:
                incidencia = response.json()
                print("\n✅ ¡Incidencia creada exitosamente!")
                print(f"📋 ID: {incidencia['id']}")
                print(f"📌 Título: {incidencia['titulo']}")
                print(f"📁 Categoría: {incidencia['categoria']}")
                print(f"⚡ Prioridad: {incidencia['prioridad']}")
                print(f"📊 Estado: {incidencia['estado']}")
            else:
                print("\n❌ Error al crear la incidencia")
        
        except Exception as e:
            print(f"\n❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")
    
    def ver_mis_incidencias(self):
        self.mostrar_encabezado("Mis Incidencias")
        
        try:
            response = self.session.get(f"{SERVER_URL}/api/incidencias",
                params={"user_id": self.usuario['id'], "user_role": "cliente"})
            
            incidencias = response.json()
            
            if not incidencias:
                print("📭 No tienes incidencias registradas\n")
            else:
                for inc in incidencias:
                    self.mostrar_incidencia_resumida(inc)
                    print()
        
        except Exception as e:
            print(f"❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")
    
    def mostrar_incidencia_resumida(self, inc):
        # Colores según estado
        estado_simbolo = {
            "Abierto": "🔵",
            "En Proceso": "🟡",
            "Pendiente Cliente": "🟣",
            "Resuelto": "🟢",
            "Cerrado": "⚫"
        }
        
        print("-" * 60)
        print(f"ID: {inc['id']} | {estado_simbolo.get(inc['estado'], '⚪')} {inc['estado']}")
        print(f"📌 {inc['titulo']}")
        print(f"📁 {inc['categoria']} | ⚡ {inc['prioridad']}")
        print(f"📅 Creada: {inc['fecha_creacion']}")
        if inc['tecnico_nombre']:
            print(f"🔧 Técnico: {inc['tecnico_nombre']}")
        else:
            print("🔧 Sin asignar")
    
    def ver_detalle_incidencia(self):
        self.mostrar_encabezado("Detalle de Incidencia")
        
        inc_id = input("Ingresa el ID de la incidencia: ").strip()
        
        if not inc_id.isdigit():
            print("\n❌ ID no válido")
            input("\nPresiona Enter para continuar...")
            return
        
        try:
            response = self.session.get(f"{SERVER_URL}/api/incidencias/{inc_id}")
            
            if response.status_code == 404:
                print("\n❌ Incidencia no encontrada")
            elif response.status_code == 200:
                inc = response.json()
                
                print(f"\n{'='*60}")
                print(f"INCIDENCIA #{inc['id']}")
                print(f"{'='*60}")
                print(f"📌 Título: {inc['titulo']}")
                print(f"📊 Estado: {inc['estado']}")
                print(f"⚡ Prioridad: {inc['prioridad']}")
                print(f"📁 Categoría: {inc['categoria']}")
                print(f"\n📄 Descripción:")
                print(f"   {inc['descripcion']}")
                print(f"\n📅 Fecha creación: {inc['fecha_creacion']}")
                print(f"🔄 Última actualización: {inc['fecha_actualizacion']}")
                
                if inc['tecnico_nombre']:
                    print(f"🔧 Técnico asignado: {inc['tecnico_nombre']}")
                else:
                    print("🔧 Sin técnico asignado")
                
                # Mostrar comentarios
                if inc.get('comentarios'):
                    print(f"\n💬 COMENTARIOS ({len(inc['comentarios'])}):")
                    print("-" * 60)
                    for com in inc['comentarios']:
                        print(f"\n👤 {com['usuario_nombre']} ({com['usuario_rol']})")
                        print(f"📅 {com['fecha']}")
                        print(f"💭 {com['comentario']}")
                        print("-" * 60)
                else:
                    print("\n💬 Sin comentarios")
        
        except Exception as e:
            print(f"\n❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")
    
    def añadir_comentario(self):
        self.mostrar_encabezado("Añadir Comentario")
        
        inc_id = input("Ingresa el ID de la incidencia: ").strip()
        
        if not inc_id.isdigit():
            print("\n❌ ID no válido")
            input("\nPresiona Enter para continuar...")
            return
        
        comentario = input("\n💬 Escribe tu comentario: ").strip()
        
        if not comentario:
            print("\n❌ El comentario no puede estar vacío")
            input("\nPresiona Enter para continuar...")
            return
        
        datos = {
            "usuario_id": self.usuario['id'],
            "usuario_nombre": self.usuario['nombre'],
            "usuario_rol": self.usuario['rol'],
            "comentario": comentario
        }
        
        try:
            response = self.session.post(
                f"{SERVER_URL}/api/incidencias/{inc_id}/comentarios", 
                json=datos
            )
            
            if response.status_code == 201:
                print("\n✅ Comentario añadido exitosamente")
            else:
                print("\n❌ Error al añadir comentario")
        
        except Exception as e:
            print(f"\n❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")

def main():
    cliente = ClienteMastertech()
    
    if cliente.login():
        cliente.menu_principal()

if __name__ == "__main__":
    main()
