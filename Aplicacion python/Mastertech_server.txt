# Rutas de archivos
DATA_DIR = 'data'

COMENTARIOS_FILE = os.path.join(DATA_DIR, 'comentarios.json')

# Crear directorio si no existe
os.makedirs(DATA_DIR, exist_ok=True)

def cargar_json(archivo):
    if os.path.exists(archivo):
        with open(archivo, 'r', encoding='utf-8') as f:
            return json.load(f)
    return []

def guardar_json(archivo, datos):
    with open(archivo, 'w', encoding='utf-8') as f:
        json.dump(datos, f, indent=2, ensure_ascii=False)

# ============ ENDPOINTS DE AUTENTICACIÓN ============

@app.route('/api/login', methods=['POST'])
def login():
    data = request.json
    usuarios = cargar_json(USUARIOS_FILE)
    
    for usuario in usuarios:
        if usuario['email'] == data['email'] and usuario['password'] == data['password']:
            return jsonify({
                "success": True,
                "usuario": {
                    "id": usuario['id'],
                    "nombre": usuario['nombre'],
                    "email": usuario['email'],
                    "rol": usuario['rol']
                }
            })
    
    return jsonify({"success": False, "error": "Credenciales incorrectas"}), 401

# ============ ENDPOINTS DE INCIDENCIAS ============

@app.route('/api/incidencias', methods=['GET'])
def get_incidencias():
    incidencias = cargar_json(INCIDENCIAS_FILE)
    
    # Filtrar por usuario si se proporciona
    user_id = request.args.get('user_id', type=int)
    user_role = request.args.get('user_role')
    
    if user_role == 'cliente' and user_id:
        incidencias = [i for i in incidencias if i['cliente_id'] == user_id]
    elif user_role == 'tecnico' and user_id:
        # Técnico ve sus asignadas + sin asignar
        incidencias = [i for i in incidencias if i.get('tecnico_id') == user_id or i.get('tecnico_id') is None]
    
    return jsonify(incidencias)

@app.route('/api/incidencias/<int:id>', methods=['GET'])
def get_incidencia(id):
    incidencias = cargar_json(INCIDENCIAS_FILE)
    incidencia = next((i for i in incidencias if i['id'] == id), None)
    
    if not incidencia:
        return jsonify({"error": "Incidencia no encontrada"}), 404
    
    # Agregar comentarios
    comentarios = cargar_json(COMENTARIOS_FILE)
    incidencia['comentarios'] = [c for c in comentarios if c['incidencia_id'] == id]
    
    return jsonify(incidencia)

@app.route('/api/incidencias', methods=['POST'])
def crear_incidencia():
    data = request.json
    incidencias = cargar_json(INCIDENCIAS_FILE)
    
    nuevo_id = max([i['id'] for i in incidencias], default=0) + 1
    
    nueva_incidencia = {
        "id": nuevo_id,
        "titulo": data['titulo'],
        "descripcion": data['descripcion'],
        "categoria": data['categoria'],
        "prioridad": data['prioridad'],
        "estado": "Abierto",
        "cliente_id": data['cliente_id'],
        "cliente_nombre": data['cliente_nombre'],
        "tecnico_id": None,
        "tecnico_nombre": None,
        "fecha_creacion": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
        "fecha_actualizacion": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    }
    
    incidencias.append(nueva_incidencia)
    guardar_json(INCIDENCIAS_FILE, incidencias)
    
    return jsonify(nueva_incidencia), 201

@app.route('/api/incidencias/<int:id>', methods=['PUT'])
def actualizar_incidencia(id):
    data = request.json
    incidencias = cargar_json(INCIDENCIAS_FILE)
    
    incidencia = next((i for i in incidencias if i['id'] == id), None)
    if not incidencia:
        return jsonify({"error": "Incidencia no encontrada"}), 404
    
    # Actualizar campos permitidos
    if 'tecnico_id' in data:
        usuarios = cargar_json(USUARIOS_FILE)
        if data['tecnico_id']:
            tecnico = next((u for u in usuarios if u['id'] == data['tecnico_id']), None)
            if tecnico:
                incidencia['tecnico_id'] = tecnico['id']
                incidencia['tecnico_nombre'] = tecnico['nombre']
                if incidencia['estado'] == 'Abierto':
                    incidencia['estado'] = 'En Proceso'
        else:
            incidencia['tecnico_id'] = None
            incidencia['tecnico_nombre'] = None
    
    if 'estado' in data:
        incidencia['estado'] = data['estado']
    
    incidencia['fecha_actualizacion'] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    guardar_json(INCIDENCIAS_FILE, incidencias)
    
    return jsonify(incidencia)

@app.route('/api/incidencias/<int:id>/comentarios', methods=['POST'])
def agregar_comentario(id):
    data = request.json
    comentarios = cargar_json(COMENTARIOS_FILE)
    
    nuevo_id = max([c['id'] for c in comentarios], default=0) + 1
    
    nuevo_comentario = {
        "id": nuevo_id,
        "incidencia_id": id,
        "usuario_id": data['usuario_id'],
        "usuario_nombre": data['usuario_nombre'],
        "usuario_rol": data['usuario_rol'],
        "comentario": data['comentario'],
        "fecha": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    }
    
    comentarios.append(nuevo_comentario)
    guardar_json(COMENTARIOS_FILE, comentarios)
    
    return jsonify(nuevo_comentario), 201

@app.route('/api/usuarios', methods=['GET'])
def get_usuarios():
    usuarios = cargar_json(USUARIOS_FILE)
    # No enviar contraseñas
    for usuario in usuarios:
        usuario.pop('password', None)
    return jsonify(usuarios)

@app.route('/api/tecnicos', methods=['GET'])
def get_tecnicos():
    usuarios = cargar_json(USUARIOS_FILE)
    tecnicos = [u for u in usuarios if u['rol'] == 'tecnico']
    for tecnico in tecnicos:
        tecnico.pop('password', None)
    return jsonify(tecnicos)

@app.route('/api/estadisticas', methods=['GET'])
def get_estadisticas():
    incidencias = cargar_json(INCIDENCIAS_FILE)

        "altas": len([i for i in incidencias if i['prioridad'] == 'Alta']),

        "total_clientes": len([u for u in usuarios if u['rol'] == 'cliente'])
    }
    
    return jsonify(estadisticas)
