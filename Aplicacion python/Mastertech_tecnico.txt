import requests
import json
import os

# Configuración del servidor
SERVER_URL = "http://localhost:5000"  # Cambiar por tu dominio: http://tu-dominio.com

class TecnicoMastertech:
    def __init__(self):
        self.usuario = None
        self.session = requests.Session()
    
    def limpiar_pantalla(self):
        os.system('clear' if os.name != 'nt' else 'cls')
    
    def mostrar_encabezado(self, titulo):
        self.limpiar_pantalla()
        print("=" * 60)
        print(f"🔧 MASTERTECH - {titulo}")
        print("=" * 60)
        if self.usuario:
            print(f"👤 Técnico: {self.usuario['nombre']} ({self.usuario['email']})")
            print("=" * 60)
        print()
    
    def login(self):
        while True:
            self.mostrar_encabezado("Inicio de Sesión - TÉCNICO")
            
            print("Ingrese sus credenciales:")
            email = input("📧 Email: ").strip()
            password = input("🔒 Contraseña: ").strip()
            
            try:
                response = self.session.post(f"{SERVER_URL}/api/login", 
                    json={"email": email, "password": password})
                
                data = response.json()
                
                if data.get('success') and data['usuario']['rol'] == 'tecnico':
                    self.usuario = data['usuario']
                    print(f"\n✅ Bienvenido {self.usuario['nombre']}!")
                    input("\nPresiona Enter para continuar...")
                    return True
                else:
                    print("\n❌ Error: Credenciales incorrectas o no es un técnico")
                    input("\nPresiona Enter para intentar de nuevo...")
            
            except requests.exceptions.ConnectionError:
                print(f"\n❌ Error: No se puede conectar al servidor en {SERVER_URL}")
                print("Verifica que el servidor esté ejecutándose.")
                input("\nPresiona Enter para intentar de nuevo...")
            except Exception as e:
                print(f"\n❌ Error: {e}")
                input("\nPresiona Enter para intentar de nuevo...")
    
    def menu_principal(self):
        while True:
            self.mostrar_encabezado("Menú Principal")
            
            print("1. 📋 Ver Mis Incidencias Asignadas")
            print("2. 📊 Ver Todas las Incidencias")
            print("3. 🔍 Ver Detalle de Incidencia")
            print("4. 🔄 Cambiar Estado de Incidencia")
            print("5. 💬 Añadir Comentario")
            print("6. 🚪 Salir")
            print()
            
            opcion = input("Selecciona una opción: ").strip()
            
            if opcion == '1':
                self.ver_mis_incidencias()
            elif opcion == '2':
                self.ver_todas_incidencias()
            elif opcion == '3':
                self.ver_detalle_incidencia()
            elif opcion == '4':
                self.cambiar_estado()
            elif opcion == '5':
                self.añadir_comentario()
            elif opcion == '6':
                print("\n👋 ¡Hasta pronto!")
                break
            else:
                print("\n❌ Opción no válida")
                input("\nPresiona Enter para continuar...")
    
    def ver_mis_incidencias(self):
        self.mostrar_encabezado("Mis Incidencias Asignadas")
        
        try:
            response = self.session.get(f"{SERVER_URL}/api/incidencias",
                params={"user_id": self.usuario['id'], "user_role": "tecnico"})
            
            incidencias = response.json()
            asignadas = [i for i in incidencias if i.get('tecnico_id') == self.usuario['id']]
            
            if not asignadas:
                print("📭 No tienes incidencias asignadas\n")
            else:
                print(f"📊 Total de incidencias asignadas: {len(asignadas)}\n")
                for inc in asignadas:
                    self.mostrar_incidencia_resumida(inc)
                    print()
        
        except Exception as e:
            print(f"❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")
    
    def ver_todas_incidencias(self):
        self.mostrar_encabezado("Todas las Incidencias")
        
        try:
            response = self.session.get(f"{SERVER_URL}/api/incidencias",
                params={"user_id": self.usuario['id'], "user_role": "tecnico"})
            
            incidencias = response.json()
            
            if not incidencias:
                print("📭 No hay incidencias en el sistema\n")
            else:
                # Separar por estado
                sin_asignar = [i for i in incidencias if i.get('tecnico_id') is None]
                asignadas_otros = [i for i in incidencias if i.get('tecnico_id') and i.get('tecnico_id') != self.usuario['id']]
                mis_asignadas = [i for i in incidencias if i.get('tecnico_id') == self.usuario['id']]
                
                print(f"📊 RESUMEN:")
                print(f"   🆕 Sin asignar: {len(sin_asignar)}")
                print(f"   ✅ Mis incidencias: {len(mis_asignadas)}")
                print(f"   👥 Asignadas a otros: {len(asignadas_otros)}")
                print()
                
                if sin_asignar:
                    print("\n🆕 INCIDENCIAS SIN ASIGNAR:")
                    print("-" * 60)
                    for inc in sin_asignar:
                        self.mostrar_incidencia_resumida(inc)
                        print()
                
                if mis_asignadas:
                    print("\n✅ MIS INCIDENCIAS:")
                    print("-" * 60)
                    for inc in mis_asignadas:
                        self.mostrar_incidencia_resumida(inc)
                        print()
        
        except Exception as e:
            print(f"❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")
    
    def mostrar_incidencia_resumida(self, inc):
        # Símbolos según estado
        estado_simbolo = {
            "Abierto": "🔵",
            "En Proceso": "🟡",
            "Pendiente Cliente": "🟣",
            "Resuelto": "🟢",
            "Cerrado": "⚫"
        }
        
        prioridad_simbolo = {
            "Crítica": "🔴",
            "Alta": "🟠",
            "Media": "🟡",
            "Baja": "🟢"
        }
        
        print("-" * 60)
        print(f"ID: {inc['id']} | {estado_simbolo.get(inc['estado'], '⚪')} {inc['estado']}")
        print(f"📌 {inc['titulo']}")
        print(f"📁 {inc['categoria']} | {prioridad_simbolo.get(inc['prioridad'], '⚪')} {inc['prioridad']}")
        print(f"👤 Cliente: {inc['cliente_nombre']}")
        print(f"📅 Creada: {inc['fecha_creacion']}")
        if inc['tecnico_nombre']:
            print(f"🔧 Técnico: {inc['tecnico_nombre']}")
        else:
            print("🔧 ⚠️ Sin asignar")
    
    def ver_detalle_incidencia(self):
        self.mostrar_encabezado("Detalle de Incidencia")
        
        inc_id = input("Ingresa el ID de la incidencia: ").strip()
        
        if not inc_id.isdigit():
            print("\n❌ ID no válido")
            input("\nPresiona Enter para continuar...")
            return
        
        try:
            response = self.session.get(f"{SERVER_URL}/api/incidencias/{inc_id}")
            
            if response.status_code == 404:
                print("\n❌ Incidencia no encontrada")
            elif response.status_code == 200:
                inc = response.json()
                
                print(f"\n{'='*60}")
                print(f"INCIDENCIA #{inc['id']}")
                print(f"{'='*60}")
                print(f"📌 Título: {inc['titulo']}")
                print(f"📊 Estado: {inc['estado']}")
                print(f"⚡ Prioridad: {inc['prioridad']}")
                print(f"📁 Categoría: {inc['categoria']}")
                print(f"👤 Cliente: {inc['cliente_nombre']}")
                print(f"\n📄 Descripción:")
                print(f"   {inc['descripcion']}")
                print(f"\n📅 Fecha creación: {inc['fecha_creacion']}")
                print(f"🔄 Última actualización: {inc['fecha_actualizacion']}")
                
                if inc['tecnico_nombre']:
                    print(f"🔧 Técnico asignado: {inc['tecnico_nombre']}")
                    if inc['tecnico_id'] == self.usuario['id']:
                        print("   ✅ (Asignada a ti)")
                else:
                    print("🔧 ⚠️ Sin técnico asignado")
                
                # Mostrar comentarios
                if inc.get('comentarios'):
                    print(f"\n💬 COMENTARIOS ({len(inc['comentarios'])}):")
                    print("-" * 60)
                    for com in inc['comentarios']:
                        print(f"\n👤 {com['usuario_nombre']} ({com['usuario_rol']})")
                        print(f"📅 {com['fecha']}")
                        print(f"💭 {com['comentario']}")
                        print("-" * 60)
                else:
                    print("\n💬 Sin comentarios")
        
        except Exception as e:
            print(f"\n❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")
    
    def cambiar_estado(self):
        self.mostrar_encabezado("Cambiar Estado de Incidencia")
        
        inc_id = input("Ingresa el ID de la incidencia: ").strip()
        
        if not inc_id.isdigit():
            print("\n❌ ID no válido")
            input("\nPresiona Enter para continuar...")
            return
        
        # Verificar que la incidencia esté asignada al técnico
        try:
            response = self.session.get(f"{SERVER_URL}/api/incidencias/{inc_id}")
            
            if response.status_code == 404:
                print("\n❌ Incidencia no encontrada")
                input("\nPresiona Enter para continuar...")
                return
            
            inc = response.json()
            
            if inc['tecnico_id'] != self.usuario['id']:
                print(f"\n❌ Esta incidencia no está asignada a ti")
                print(f"   Técnico asignado: {inc['tecnico_nombre'] or 'Sin asignar'}")
                input("\nPresiona Enter para continuar...")
                return
            
            print(f"\nIncidencia: {inc['titulo']}")
            print(f"Estado actual: {inc['estado']}")
            print("\n📊 Estados disponibles:")
            estados = ["En Proceso", "Pendiente Cliente", "Resuelto"]
            
            for i, estado in enumerate(estados, 1):
                print(f"  {i}. {estado}")
            
            estado_num = input("\nSelecciona nuevo estado (1-3): ").strip()
            
            if not estado_num.isdigit() or int(estado_num) < 1 or int(estado_num) > 3:
                print("\n❌ Estado no válido")
                input("\nPresiona Enter para continuar...")
                return
            
            nuevo_estado = estados[int(estado_num) - 1]
            
            # Actualizar estado
            response = self.session.put(
                f"{SERVER_URL}/api/incidencias/{inc_id}",
                json={"estado": nuevo_estado}
            )
            
            if response.status_code == 200:
                print(f"\n✅ Estado actualizado a: {nuevo_estado}")
            else:
                print("\n❌ Error al actualizar el estado")
        
        except Exception as e:
            print(f"\n❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")
    
    def añadir_comentario(self):
        self.mostrar_encabezado("Añadir Comentario")
        
        inc_id = input("Ingresa el ID de la incidencia: ").strip()
        
        if not inc_id.isdigit():
            print("\n❌ ID no válido")
            input("\nPresiona Enter para continuar...")
            return
        
        # Verificar que esté asignada al técnico
        try:
            response = self.session.get(f"{SERVER_URL}/api/incidencias/{inc_id}")
            
            if response.status_code == 404:
                print("\n❌ Incidencia no encontrada")
                input("\nPresiona Enter para continuar...")
                return
            
            inc = response.json()
            
            if inc['tecnico_id'] != self.usuario['id']:
                print(f"\n❌ Solo puedes comentar en incidencias asignadas a ti")
                input("\nPresiona Enter para continuar...")
                return
            
            print(f"\nIncidencia: {inc['titulo']}")
            comentario = input("\n💬 Escribe tu comentario: ").strip()
            
            if not comentario:
                print("\n❌ El comentario no puede estar vacío")
                input("\nPresiona Enter para continuar...")
                return
            
            datos = {
                "usuario_id": self.usuario['id'],
                "usuario_nombre": self.usuario['nombre'],
                "usuario_rol": self.usuario['rol'],
                "comentario": comentario
            }
            
            response = self.session.post(
                f"{SERVER_URL}/api/incidencias/{inc_id}/comentarios", 
                json=datos
            )
            
            if response.status_code == 201:
                print("\n✅ Comentario añadido exitosamente")
            else:
                print("\n❌ Error al añadir comentario")
        
        except Exception as e:
            print(f"\n❌ Error: {e}")
        
        input("\nPresiona Enter para continuar...")

def main():
    tecnico = TecnicoMastertech()
    
    if tecnico.login():
        tecnico.menu_principal()

if __name__ == "__main__":
    main()
